#[20. 有效的括号](https://leetcode-cn.com/problems/valid-parentheses/)
#思路：利用栈，遍历数组，碰到左括号，就进行入栈，碰到又括号进行匹配；
```
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == '(')
                st.push(')');
            else if (s[i] == '[') 
                st.push(']');
            else if (s[i] == '{') 
                st.push('}');
            else if(st.empty() || st.top() != s[i])
                return false;
            else
                st.pop();
        }
        if(st.empty())
            return true;
        return false;
    }
};
```

#[155. 最小栈](https://leetcode-cn.com/problems/min-stack/)
#思路：两个站，一个站存储value，一个站存储最小值；
```
class MinStack {
public:
    /** initialize your data structure here. */
    MinStack() {   
    }
    void push(int x) {
        st1.push(x);
        if (st2.empty())
            st2.push(x);
        else {
            int a = st2.top();
            if (a < x)
                st2.push(a);
            else
                st2.push(x);
        }
    }
    void pop() {
        if (!st1.empty()) {
            st1.pop();
            st2.pop();
        }
    }
    int top() {
        return st1.top();
    }
    int getMin() {
        return st2.top();
    }
    stack<int> st1;
    stack<int> st2;
};
```
#[84. 柱状图中最大的矩形](https://leetcode-cn.com/problems/largest-rectangle-in-histogram/)
#思路1：暴力破解，两层循环(超时)；
#思路2：辅助栈，遍历数组，如果当前位置的值大于栈顶值，则入栈，否则将栈里大于当前位置的值全部输出计算面积；
```
class Solution {
public:
    int largestRectangleArea(vector<int>& h) {
        int n = h.size();
        int res = 0;
        stack<int> st;
        st.push(-1);
        for (int i = 0; i < n; i++) {
            while (st.size() > 1 && h[st.top()] > h[i]) {
                int a = st.top();
                st.pop();
                int b = st.top();
                int area = h[a] * (i - b - 1);
                res = max(res, area);
            }
            st.push(i);
        }
        while (st.size() > 1) {
            int a = st.top();
            st.pop();
            int b = st.top();
            int area = h[a] * (n - b - 1);
            res = max(res, area);
        }
        return res;
    }
};
```
#思路3：利用两个数组，分别存储每个节点的左边界和右边界，然后计算面积最大值；
```
class Solution {
public:
    int largestRectangleArea(vector<int>& h) {
        if (h.empty()) return 0;
        int n = h.size();
        vector<int> l(n, -1), r(n, n);
        for (int i = 1; i < n; ++i) {
            int p = i - 1;
            for (; p >= 0 && h[p] >= h[i]; p = l[p]);
            l[i] = p;
        }
        for (int i = n - 2; i >= 0; --i) {
            int p = i + 1;
            for (; p < n && h[p] >= h[i]; p = r[p]);
            r[i] = p;
        }
        int a = 0;
        for (int i = 0; i < n; ++i) a = max(a, h[i] * (r[i] - l[i] - 1));
        return a;
    }
};
```

#[239. 滑动窗口最大值](https://leetcode-cn.com/problems/sliding-window-maximum/)
##思路：双端队列，主要用到了双端队列可以从两端进行pop，遍历一遍数组，判断当前节点的值与队列末尾值的大小，如果大于或等于，则将队列中小于等于节点值的节点从后pop出队列，然后将该节点push入队列，每次将队列的front值输出到结果数组；
```
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        if (k == 1)
            return nums;
        vector<int> res;
        deque<int> dq;
        int n = nums.size();
        for (int i = 0; i < k && i < n; i++) {
            while (!dq.empty() && nums[i] >= nums[dq.back()])
                dq.pop_back();
            dq.push_back(i);
        }
        res.push_back(nums[dq.front()]);
        for (int i = k; i < n; i++) {
            if (!dq.empty() && dq.front() <= i - k)
                dq.pop_front();
            while (!dq.empty() && nums[i] >= nums[dq.back()])
                dq.pop_back();
            dq.push_back(i);
            res.push_back(nums[dq.front()]);
        }
        return res;
    }
};
```
#[641. 设计循环双端队列](https://leetcode-cn.com/problems/design-circular-deque/)
#思路：new一个数组，双指针分别指向队头和队尾，count记录存储值的数量，len记录数组最大长度；
```
class MyCircularDeque {
public:
    /** Initialize your data structure here. Set the size of the deque to be k. */
    MyCircularDeque(int k) {
        nums = new int[k];
        i = 0;
        j = 0;
        len = k;
        count = 0;
    }    
    /** Adds an item at the front of Deque. Return true if the operation is successful. */
    bool insertFront(int value) {
        if (count == len)
            return false;
        j = (j + len - 1) % len;
        nums[j] = value;
        count++;
        return true;
    }   
    /** Adds an item at the rear of Deque. Return true if the operation is successful. */
    bool insertLast(int value) {
        if (count == len)
            return false;
        nums[i] = value;
        i = (i + 1) % len;
        count++;
        return true;
    }    
    /** Deletes an item from the front of Deque. Return true if the operation is successful. */
    bool deleteFront() {
        if (count == 0)
            return false;
        j = (j + 1) % len;
        count--;
        return true;
    }    
    /** Deletes an item from the rear of Deque. Return true if the operation is successful. */
    bool deleteLast() {
        if (count == 0)
            return false;
        i = (i + len - 1) % len;
        count--;
        return true;
    }    
    /** Get the front item from the deque. */
    int getFront() {
        if (count == 0)
            return -1;
        return nums[j];
    }   
    /** Get the last item from the deque. */
    int getRear() {
        if (count == 0)
            return -1;
        return nums[(i + len - 1) % len];
    }    
    /** Checks whether the circular deque is empty or not. */
    bool isEmpty() {
        if (count == 0)
            return true;
        return false;
    }   
    /** Checks whether the circular deque is full or not. */
    bool isFull() {
        if (count == len)
            return true;
        return false;
    }
    private:
    int *nums;
    int i;
    int j;
    int len;
    int count;
};
```

#[42. 接雨水](https://leetcode-cn.com/problems/trapping-rain-water/)
##思路1：双指针，这里需要曾加左最大值和右最大值，当左指针指向的位置值小于右指针指向的位置值时，首先判断该值是否大于左最大值，大于，则将值赋给左最大值，小于则计算盛水量，同理计算右指针小于左指针；
```
class Solution {
public:
    int trap(vector<int>& h) {
        int res = 0;
        int left = 0;
        int right = h.size() - 1;
        int left_max = 0;
        int right_max = 0;
        while (left < right) {
            if (h[left] <= h[right]) {
                if (h[left] >= left_max)
                    left_max = h[left];
                else
                    res += left_max - h[left];
                left++;
            }
            else {
                if(h[right] >= right_max)
                    right_max = h[right];
                else
                    res += right_max - h[right];
                right--;
            }
        }
        return res;
    }
};
```
#思路2：用栈作为辅助，遍历数组，当前位置的值大于数组顶端的值时，将栈中所有小于该值的节点出栈，计算盛水量；
```
class Solution {
public:
    int trap(vector<int>& h) {
        int res = 0;
        stack<int> st;
        for (int i = 0; i < h.size(); i++) {
            while (!st.empty() && h[i] > h[st.top()]) {
                int a = st.top();
                st.pop();
                if (st.empty())
                    break;
                int b = min(h[st.top()], h[i]) - h[a];
                int c = i - st.top() - 1;
                res += b * c;
            }
            st.push(i);
        }
        return res;
    }
};
```
